def add_and_maybe_multiply(a,b,c=None):
    
    if(c != None):
        x = (a+b)*c
    else:
        x = a+b
    
    return x

while(x > 0):
    if(total>500): #total이 500이 넘으면 while문 탈출
        break
    total += x  #total은 while문이 한 번 돌때마다 x와 더해진 값으로 저장 
    x = x//2  #x는 while문이 힌 번 돌때마다 2로 나눈 몫으로 저장
